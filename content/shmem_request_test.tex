\apisummary{
    The request test routine provides a method for querying the status
    of the request.
}

\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_test(shmem_request_t *req, int *status);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN}{req}{Handle to the request object.}
    \apiargument{OUT}{status}{The status of the request.}
\end{apiarguments}

\apidescription{
    The routine returns zero via the status argument if the request is
    finished, otherwise a non zero value is returned.
    It is safe to free the request object, if the return value is zero.
 }

\apireturnvalues{
    None.
}

\apinotes{ None }

\end{apidefinition}
