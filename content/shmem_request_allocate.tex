\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_allocate(shmem_request_t *req);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN/OUT}{req}{Handle to the request object.}
\end{apiarguments}

\apidescription{
    This routine can be used to explicitly pre-allocate a request
    object. On failure the handle will be set to SHMEM\_REQUEST\_INVALID.
 }

\apireturnvalues{
    None.
}

\apinotes{ None }

\end{apidefinition}
