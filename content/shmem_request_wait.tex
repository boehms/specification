\apisummary{
    The request wait routine provides a method for completing
    outstanding operations on a request.
}

\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_wait(shmem_request_t *req);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN}{req}{The request handle to wait for.}
\end{apiarguments}

\apidescription{
    The routine blocks until all outstanding operations on the request
    are finished. Outstanding operations are operations posted before
    wait was called. After the routine returns, the request is set to
    NULL. There are no guarantees on the ordering of the outstanding
    operations.
 }

\apireturnvalues{
    None.
}

\apinotes{ In a threaded environment, calling shmem\_request\_wait from
a thread other than the thread, posting the operation is permitted.
However, care must be taken that no operations are posted while the
request is being waited on. }

\end{apidefinition}
