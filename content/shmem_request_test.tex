\apisummary{
    The request test routine provides a method for querying the status
    of the request.
}

\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_test(shmem_request_t *req, int *status);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN}{req}{The request handle to test the status.}
    \apiargument{OUT}{status}{The status of the request.}
\end{apiarguments}

\apidescription{
    The routine returns zero if the request is finished, otherwise a non
    zero value is returned. If the return value is 0, it is safe to free
    the request object.
 }

\apireturnvalues{
    None.
}

\apinotes{ None }

\end{apidefinition}
